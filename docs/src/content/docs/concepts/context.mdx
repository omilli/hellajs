---
title: Context
description: Share state across components using simple closures.
---

HellaJS implements context patterns using JavaScript closures and the [`slot`](/packages/dom/slot) function.

```jsx
import { signal } from '@hellajs/core';
import { mount, slot } from '@hellajs/dom';

const createAuth = () => {
  const user = signal(null);
  const login = (userData) => user(userData);
  const logout = () => user(null);
  
  const Provider = slot((props, children) => (
    <div {...props}>
      {children}
    </div>
  ));
  
  return { Provider, user, login, logout };
};

const { Provider, user, login, logout } = createAuth();

const LoginButton = () => (
  <button onclick={() => login({ name: 'John' })}>
    Login
  </button>
);

const UserProfile = () => (
  <div>
    {() => user() ? `Welcome, ${user().name}!` : 'Not logged in'}
  </div>
);

const App = () => (
  <Provider class="auth-provider">
    <LoginButton />
    <UserProfile />
    <button onclick={logout}>Logout</button>
  </Provider>
);

mount(App, '#app');
```

## Multiple Contexts

```jsx
const Auth = createAuth();
const Theme = createTheme();

const App = () => (
  <Auth.Provider>
    <Theme.Provider>
      <p>User: {() => Auth.user()?.name || 'Guest'}</p>
      <p>Theme: {() => Theme.current}</p>
    </Theme.Provider>
  </Auth.Provider>
);
```
