Creates reusable component patterns with props and children.

```jsx
import { slot } from '@hellajs/dom';

const Card = slot((props, children) => (
  <div class={`card ${props.variant || 'default'}`}>
    {children}
  </div>
));

const App = () => (
  <Card variant="elevated">
    <h2>Card Title</h2>
    <p>Card content goes here</p>
  </Card>
);
```

## API

```typescript
function slot<T = {}>(
  render: (props: T, children: VNodeValue[]) => VNode
): (props: T & { children?: VNodeValue[] }) => VNode
```

## Context Patterns

Create provider components that share state or configuration with their children.

```jsx
import { signal } from '@hellajs/core';
import { slot } from '@hellajs/dom';

const createTheme = () => {
  const theme = signal('light');
  
  const Provider = slot((props, children) => (
    <div class={`theme-${theme()}`} {...props}>
      {children}
    </div>
  ));
  
  return { Provider, theme };
};

const { Provider, theme } = createTheme();

const App = () => (
  <Provider>
    <p>Current theme: {theme()}</p>
    <button onclick={() => theme(theme() === 'light' ? 'dark' : 'light')}>
      Toggle
    </button>
  </Provider>
);
```
