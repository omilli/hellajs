---
title: untracked
description: Read signals inside effects without creating a dependency.
---

Read [signals](/packages/core/signal) inside effects without creating a dependency.

```typescript
import { signal, effect, untracked } from '@hellajs/core';

const count = signal(0);
const threshold = signal(10);

effect(() => {
  // This creates a dependency on count
  const currentCount = count();

  // This does NOT create a dependency on threshold
  const currentThreshold = untracked(() => threshold());

  console.log(`Count: ${currentCount}, Threshold: ${currentThreshold}`);
});

// Effect runs, logs: "Count: 0, Threshold: 10"

count(1);
// Effect runs, logs: "Count: 1, Threshold: 10"

threshold(20);
// Effect doesn't run, since threshold was read via untracked
```

## API

```typescript
function untracked<T>(fn: () => T): T
```

## Untracked vs Tracked

Compare how tracked and untracked signal reads affect effect dependencies.

```typescript
const a = signal(1);
const b = signal(2);

// Tracked: Creates dependencies on both a and b
effect(() => {
  console.log('Tracked:', a(), b());
}); // Runs when either a or b changes

// Partially untracked: Only depends on a
effect(() => {
  const valueA = a();
  const valueB = untracked(() => b());
  console.log('Partial:', valueA, valueB);
}); // Only runs when a changes

// Fully untracked: No dependencies
effect(() => {
  untracked(() => {
    console.log('Untracked:', a(), b());
  });
}); // Runs only once (initial)
```
