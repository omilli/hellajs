## Theme Provider

Create a theme system that can be switched dynamically:

```jsx
import { signal, effect } from '@hellajs/core';
import { cssVars } from '@hellajs/css';

const createThemeSystem = () => {
  const currentTheme = signal('light');
  
  const themes = {
    light: {
      colors: {
        background: '#ffffff',
        surface: '#f8f9fa',
        text: '#212529',
        textSecondary: '#6c757d',
        border: '#dee2e6'
      }
    },
    
    dark: {
      colors: {
        background: '#121212',
        surface: '#1e1e1e',
        text: '#ffffff',
        textSecondary: '#adb5bd',
        border: '#495057'
      }
    },
    
    blue: {
      colors: {
        background: '#0d1117',
        surface: '#161b22',
        text: '#c9d1d9',
        textSecondary: '#8b949e',
        border: '#30363d'
      }
    }
  };

  // Apply theme when changed
  effect(() => {
    const theme = themes[currentTheme()];
    if (theme) {
      cssVars(theme);
      document.documentElement.setAttribute('data-theme', currentTheme());
    }
  });

  return {
    currentTheme,
    setTheme: (theme) => currentTheme(theme),
    getAvailableThemes: () => Object.keys(themes)
  };
};

const theme = createThemeSystem();

// Theme switcher component
const ThemeSwitcher = () => {
  return (
    <select 
      value={theme.currentTheme()}
      onchange={(e) => theme.setTheme(e.target.value)}
    >
      {theme.getAvailableThemes().map(themeName => (
        <option key={themeName} value={themeName}>
          {themeName.charAt(0).toUpperCase() + themeName.slice(1)}
        </option>
      ))}
    </select>
  );
};
```

## Component Style Variants

Create reusable component styles with variants:

```jsx
import { css } from '@hellajs/css';

const createButtonStyles = () => {
  const base = css({
    display: 'inline-flex',
    alignItems: 'center',
    justifyContent: 'center',
    padding: 'var(--spacing-sm) var(--spacing-md)',
    borderRadius: 'var(--borders-radius-md)',
    border: 'var(--borders-width-thin) solid transparent',
    fontFamily: 'var(--typography-fontFamily-primary)',
    fontSize: 'var(--typography-fontSize-base)',
    fontWeight: '500',
    lineHeight: 'var(--typography-lineHeight-tight)',
    cursor: 'pointer',
    transition: 'all 0.2s ease',
    textDecoration: 'none',
    
    '&:focus': {
      outline: '2px solid var(--colors-primary)',
      outlineOffset: '2px'
    },
    
    '&:disabled': {
      opacity: '0.6',
      cursor: 'not-allowed',
      pointerEvents: 'none'
    }
  });

  const variants = {
    primary: css({
      backgroundColor: 'var(--colors-primary)',
      color: 'white',
      borderColor: 'var(--colors-primary)',
      
      '&:hover': {
        backgroundColor: '#0056b3',
        borderColor: '#0056b3'
      }
    }),
    
    secondary: css({
      backgroundColor: 'var(--colors-secondary)',
      color: 'white',
      borderColor: 'var(--colors-secondary)',
      
      '&:hover': {
        backgroundColor: '#545b62',
        borderColor: '#545b62'
      }
    }),
    
    outline: css({
      backgroundColor: 'transparent',
      color: 'var(--colors-primary)',
      borderColor: 'var(--colors-primary)',
      
      '&:hover': {
        backgroundColor: 'var(--colors-primary)',
        color: 'white'
      }
    }),
    
    ghost: css({
      backgroundColor: 'transparent',
      color: 'var(--colors-text)',
      border: 'none',
      
      '&:hover': {
        backgroundColor: 'var(--colors-surface)'
      }
    })
  };

  const sizes = {
    sm: css({
      padding: 'var(--spacing-xs) var(--spacing-sm)',
      fontSize: 'var(--typography-fontSize-sm)'
    }),
    
    md: css({
      padding: 'var(--spacing-sm) var(--spacing-md)',
      fontSize: 'var(--typography-fontSize-base)'
    }),
    
    lg: css({
      padding: 'var(--spacing-md) var(--spacing-lg)',
      fontSize: 'var(--typography-fontSize-lg)'
    })
  };

  return { base, variants, sizes };
};

const buttonStyles = createButtonStyles();

// Button component using style variants
const Button = ({ variant = 'primary', size = 'md', children, ...props }) => {
  const className = [
    buttonStyles.base,
    buttonStyles.variants[variant],
    buttonStyles.sizes[size]
  ].join(' ');

  return (
    <button class={className} {...props}>
      {children}
    </button>
  );
};

// Usage
const App = () => (
  <div>
    <Button variant="primary" size="sm">Primary Small</Button>
    <Button variant="secondary" size="md">Secondary Medium</Button>
    <Button variant="outline" size="lg">Outline Large</Button>
    <Button variant="ghost">Ghost Default</Button>
  </div>
);
```

## Responsive Design Patterns

Create responsive utilities and breakpoint systems:

```jsx
import { css } from '@hellajs/css';

const breakpoints = {
  sm: '640px',
  md: '768px',
  lg: '1024px',
  xl: '1280px',
  '2xl': '1536px'
};

const createResponsiveUtils = () => {
  const mediaQueries = Object.entries(breakpoints).reduce((acc, [key, value]) => {
    acc[key] = `@media (min-width: ${value})`;
    return acc;
  }, {});

  const grid = css({
    display: 'grid',
    gap: 'var(--spacing-md)',
    gridTemplateColumns: '1fr',
    
    [mediaQueries.sm]: {
      gridTemplateColumns: 'repeat(2, 1fr)'
    },
    
    [mediaQueries.md]: {
      gridTemplateColumns: 'repeat(3, 1fr)'
    },
    
    [mediaQueries.lg]: {
      gridTemplateColumns: 'repeat(4, 1fr)'
    },
    
    [mediaQueries.xl]: {
      gridTemplateColumns: 'repeat(6, 1fr)'
    }
  });

  const container = css({
    width: '100%',
    marginLeft: 'auto',
    marginRight: 'auto',
    paddingLeft: 'var(--spacing-md)',
    paddingRight: 'var(--spacing-md)',
    
    [mediaQueries.sm]: {
      maxWidth: '640px'
    },
    
    [mediaQueries.md]: {
      maxWidth: '768px'
    },
    
    [mediaQueries.lg]: {
      maxWidth: '1024px'
    },
    
    [mediaQueries.xl]: {
      maxWidth: '1280px'
    }
  });

  const spacing = {
    responsive: (property, values) => {
      const styles = {};
      Object.entries(values).forEach(([breakpoint, value]) => {
        if (breakpoint === 'base') {
          styles[property] = value;
        } else {
          styles[mediaQueries[breakpoint]] = { [property]: value };
        }
      });
      return css(styles);
    }
  };

  return { grid, container, spacing, mediaQueries };
};

const responsive = createResponsiveUtils();

// Responsive component example
const ResponsiveCard = ({ children }) => {
  const cardStyles = css({
    backgroundColor: 'var(--colors-surface)',
    borderRadius: 'var(--borders-radius-lg)',
    padding: 'var(--spacing-md)',
    boxShadow: 'var(--shadows-md)',
    border: 'var(--borders-width-thin) solid var(--colors-border)',
    
    [responsive.mediaQueries.md]: {
      padding: 'var(--spacing-lg)'
    }
  });

  return <div class={cardStyles}>{children}</div>;
};

const ResponsiveLayout = () => (
  <div class={responsive.container}>
    <div class={responsive.grid}>
      <ResponsiveCard>Card 1</ResponsiveCard>
      <ResponsiveCard>Card 2</ResponsiveCard>
      <ResponsiveCard>Card 3</ResponsiveCard>
      <ResponsiveCard>Card 4</ResponsiveCard>
    </div>
  </div>
);
```
