Define and inject CSS rules from JS objects.

```jsx
import { css } from '@hellajs/css';

const className = css({
  color: 'red',
  ':hover': {
    color: 'blue'
  },
  $span: {
    color: 'green'
  },
  '@media (max-width: 600px)': {
    color: 'purple'
  }
});
```

## API

```typescript
function css(styles: CSSObject, options?: CSSOptions): string

interface CSSObject {
  [key: string]: string | number | CSSObject;
}

interface CSSOptions {
  name?: string;        // Custom class name
  global?: boolean;     // Global styles (no class generation)
  scoped?: string;      // Scope styles to selector
}

// Static methods
css.remove(className: string): void
css.reset(): void
```

## Nesting & Selectors

Use nested syntax for pseudo-selectors, child elements, and media queries within your styles.

```jsx
const className = css({
  color: 'red',
  // Pseudo-selectors
  ':hover': {
    color: 'blue'
  },
  // Child elements
  $span: {
    color: 'green'
  },
  // Parent reference
  '&.active': {
    fontWeight: 'bold'
  },
  // Media queries
  '@media (max-width: 600px)': {
    fontSize: '14px'
  }
});
```

## Global Styles

Apply styles globally without generating class names using the global option.

```jsx
css({
  body: {
    margin: 0,
    fontFamily: 'Arial, sans-serif'
  }
}, { global: true });
```
